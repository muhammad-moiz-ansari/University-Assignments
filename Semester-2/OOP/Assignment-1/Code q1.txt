int* one_compl(int* arr, int length)
{
    for (int i = 0; i < length; ++i)
    {
        if (arr[i] == 1)
            arr[i] = 0;
        else
            arr[i] = 1;
    }
    return arr;
}

int* two_compl(int* arr, int length)
{
    bool invert = 0;

    for (int i = length; i > 0; --i)
    {
        if (invert)
        {
            if (arr[i] == 1)
                invert = 1;
            else
                invert = 0;

            if (arr[i] == 1)
                arr[i] = 0;
            else
                arr[i] = 1;
        }
    }
    return arr;
}
