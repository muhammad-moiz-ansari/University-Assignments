class CAList

    int ksize = 7, vsize = 6, k_index = 0, * v_index = new int[ksize-1];

    CAList& operator[](const char*);

    CAList& operator=(const char*);

//Default Constructor
CAList::CAList()
{
    arr = new string * [ksize];
    for (int i = 0; i < ksize; ++i)
        arr[i] = new string[vsize+1];

    for (int i = 0; i < ksize; ++i)
    {
        v_index[i] = 1;
        b_key[i] = 0;
    }
}

CAList& CAList::operator[](const char* key)
{
    k_index = 0;
    int i = 0;
    for (; i < ksize; ++i)
    {
        if (arr[i][0] == "\0")
        {
            k_index = i;
            break;
        }
        else if (arr[i][0] == key)
        {
            k_index = i;
            break;
        }
    }
    if (!(i < ksize))
    {
        cout << "Error: Key space not available. Maximum number of keys attained. The key \"" << key << "\" can not be added." << endl << endl;
    }
    else
    {
        arr[k_index][0] = key;
        b_key[k_index] = 1;
    }

    return *this;
}

CAList& CAList::operator=(const char* value)
{
    int index = -1;
    
    for(int i=0; i<ksize; ++i)
        if (b_key[k_index])
        {
            index = k_index;
            break;
        }

    if (b_key[k_index] && index>=0)
    {
        if (v_index[index] <= vsize)
        {
            arr[index][v_index[index]] = value;
            if (v_index[index] <= vsize)
                ++v_index[index];
            b_key[index] = 0;
        }
        else
        {
            cout << "Error: Value space not available. Maximum number of values attained. The value \"" << value << "\", to the"
                << " key \"" << arr[index][0] << "\" can not be added." << endl << endl;
            b_key[k_index] = 0;
        }
    }
    
    return *this;
}

ostream& operator<<(ostream& output, const CAList& list)
{
    output << "[";
    for (int i = 0; i < list.ksize; ++i)
    {
        if (list.arr[i][0] != "\0")
        {
            if (i != 0)
                output << " ";
            output << list.arr[i][0] << ":" << "{";
            for (int j = 1; j <= list.vsize; ++j)
            {
                if (list.arr[i][j] != "\0")
                    output << list.arr[i][j];
                if (list.arr[i][j + 1] != "\0" && j < list.vsize)
                    output << ", ";
                else
                    break;
            }
            output << "}";
            if (i < (list.ksize - 1) && list.arr[i + 1][0] != "\0")
                output << ", \n";
        }
    }
    output << "]";

    output << endl << endl;

    return output;
}

//Destructor
CAList::~CAList()
{
    for (int i = 0; i < ksize; ++i)
        delete[] arr[i];
    delete[] arr;
    arr = NULL;
}
